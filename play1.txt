#include<iostream>
using namespace std;
int main()
{
    int x;
    cin>>x;
    int arr[x],b[x],i,j=0;
    for(i=0;i<x;i++)
    cin>>arr[i];
    for(i=0;i<x-1;i++)
    {
        b[j++]=arr[i]|arr[i+1];
    }
    if(i==x-1)
    b[j]=arr[i]|arr[i];
    for(i=0;i<x;i++)
    cout<<b[i]<<" ";
    return 0;
}
